В программе не правильно обозначен диапазон возраста тинэйджеров(<19).
Поэтому тест на возрасте 19 лет выдает False.
Также и возраст 0, и 10 лет не являются возрастом тинэйджеров, а в программе не отмечены, тест выдает False.
Должно быть от 13 включительно и до 19 включительно(>= 13 && age <= 19).
public class Person {
    public static boolean isTeenager(int age)
    {
        boolean result;
        if (age >= 13 && age <= 19) {
            result = true;
        }
        else {
            result = false;
        }
        return result;
    }
}
